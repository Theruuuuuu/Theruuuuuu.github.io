Chart.js??

他是一個javascript的程式庫，能夠實現圖表功能。

使用方法

在你要使用的HTML，加上引入

<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

然後在你想要的地方加入

<div>
  <canvas id="myChart"></canvas>
</div>

根據官方說明文件建立一個簡單的長條圖

<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

<div>
  <canvas id="myChart"></canvas>
</div>


<script>
  const ctx = document.getElementById('myChart');

  new Chart(ctx, {
    type: 'bar',
    data: {
      labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'], 
      datasets: [{
        label: '# of Votes',
        data: [12, 19, 3, 5, 2, 3],
        borderWidth: 1
      }]
    },
    options: {
      scales: {
        y: {
          beginAtZero: true
        }
      }
    }
  });
</script>

加上Ajax 做配合

使用Ajax可以讓我們的圖表資料依據資料的不同而改變

  • 修改程式碼改為按下按鈕後取得圖表
<div>
    //加上按鈕
    <a id="get" class="btn btn-primary">圖表!</a>
    <canvas id="myChart"></canvas>
</div>

@section Scripts{
    <script>
      $(function(){
        //按鈕按下
          $("#get").click(function(){
              $.ajax({
                  type:"POST",
                  url:"YourApi",
                  data:"",
                  dataType:"json",
                  success: OnSuccessResult,
                  error: OnError
              });
              //成功取回資料就產生圖表
              function OnSuccessResult (data){
                  var _data=data;
                  var _labels = _data[0]; //資料Labels
                  var _chartValue = _data[1]; //資料列
                //顏色
                  var barColor=['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange','Brown'];

                  new Chart("myChart",{
                      type:"bar",
                      //在這裡把東西都丟進去
                      data:{
                        labels: _labels,
                        datasets:[{
                            backgroundColor: barColor,
                            data:_chartValue
                        }]
                      }
                  });
              }
              //錯誤時的處理
              function OnError (data){
                //dosomething
              }
          });
      });
    </script>
}
